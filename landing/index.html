    Write once. Run everywhere.

    Server routes and native app surfaces share the same language, concurrency model, and documentation path.

    Server - REST API
    import PrismServer

let server = PrismHTTPServer(port: 8080)

await server.get("/health") { _ in
    .json(["status": "ok"])
}

await server.group("/api") { api in
    api.get("/users") { _ in
        .json(["users": []])
    }
}

try await server.start()
    SwiftUI - Native App
    import Prism
import SwiftUI

struct CounterView: View {
    let store: PrismStore<Counter.State, Counter.Action>

    var body: some View {
        VStack {
            Text("\\(store.state.count)")
                .prismTypography(.headlineLarge)
            PrismButton("Increment") {
                store.send(.increment)
            }
        }
    }
}
